
            const badge = document.getElementById('zapiStatusBadge');
            const dot = badge.querySelector('.status-dot');
            const text = badge.querySelector('.status-text');
            
            dot.style.background = '#9ca3af';
            text.textContent = 'Verificando...';
            badge.style.background = 'rgba(156, 163, 175, 0.1)';
            badge.style.border = '1px solid rgba(156, 163, 175, 0.3)';
            
            try {
                const response = await fetch('/api/integrations/zapi/health', {
                    headers: { 'Authorization': 'Bearer ' + localStorage.getItem('access_token') }
                });
                const data = await response.json();
                
                if (data.status === 'connected' && data.can_send_messages) {
                    dot.style.background = '#10b981';
                    text.textContent = 'WhatsApp Conectado';
                    badge.style.background = 'rgba(16, 185, 129, 0.1)';
                    badge.style.border = '1px solid rgba(16, 185, 129, 0.3)';
                    badge.style.color = '#10b981';
                } else if (data.status === 'disconnected') {
                    dot.style.background = '#f59e0b';
                    text.textContent = 'WhatsApp Desconectado';
                    badge.style.background = 'rgba(245, 158, 11, 0.1)';
                    badge.style.border = '1px solid rgba(245, 158, 11, 0.3)';
                    badge.style.color = '#f59e0b';
                } else if (data.status === 'not_configured') {
                    dot.style.background = '#ef4444';
                    text.textContent = 'Z-API Nao Configurada';
                    badge.style.background = 'rgba(239, 68, 68, 0.1)';
                    badge.style.border = '1px solid rgba(239, 68, 68, 0.3)';
                    badge.style.color = '#ef4444';
                } else {
                    dot.style.background = '#ef4444';
                    text.textContent = 'Erro na Conexao';
                    badge.style.background = 'rgba(239, 68, 68, 0.1)';
                    badge.style.border = '1px solid rgba(239, 68, 68, 0.3)';
                    badge.style.color = '#ef4444';
                }
            } catch (error) {
                console.error('Erro ao verificar status Z-API:', error);
                dot.style.background = '#ef4444';
                text.textContent = 'Erro ao Verificar';
                badge.style.background = 'rgba(239, 68, 68, 0.1)';
                badge.style.border = '1px solid rgba(239, 68, 68, 0.3)';
                badge.style.color = '#ef4444';
            }
        }
