CORREÇÃO 1 — Adicionar jti (JWT ID) ao token [pré-requisito para a #2]

No código de geração do JWT (core/security.py), adicione o campo jti usando uuid4:

    import uuid
    to_encode.update({
        "exp": expire,
        "iat": now,
        "iss": settings.JWT_ISSUER,
        "aud": settings.JWT_AUDIENCE,
        "type": "access",
        "jti": str(uuid.uuid4())   # ← adicionar esta linha
    })

CORREÇÃO 2 — Implementar blacklist de tokens no logout

Crie uma tabela no banco de dados para tokens revogados:

    CREATE TABLE revoked_tokens (
        jti VARCHAR(36) PRIMARY KEY,
        revoked_at TIMESTAMP NOT NULL DEFAULT NOW(),
        expires_at TIMESTAMP NOT NULL
    );

    # Model SQLAlchemy correspondente:
    class RevokedToken(Base):
        __tablename__ = "revoked_tokens"
        jti = Column(String(36), primary_key=True)
        revoked_at = Column(DateTime, default=datetime.utcnow)
        expires_at = Column(DateTime, nullable=False)

No endpoint de logout (api/endpoints/auth.py), antes de deletar os cookies,
insira o jti do token atual na tabela:

    @router.post("/logout")
    async def logout(request: Request, response: Response,
                     current_user = Depends(get_current_user),
                     db: Session = Depends(get_db)):
        # Extrai o token do cookie
        token = request.cookies.get("access_token")
        if token:
            try:
                payload = jwt.decode(token, settings.SECRET_KEY,
                                     algorithms=[settings.ALGORITHM],
                                     audience=settings.JWT_AUDIENCE,
                                     issuer=settings.JWT_ISSUER)
                jti = payload.get("jti")
                exp = datetime.utcfromtimestamp(payload.get("exp"))
                if jti:
                    db.add(RevokedToken(jti=jti, expires_at=exp))
                    db.commit()
            except Exception:
                pass  # token inválido já não representa risco
        record_security_event("logout", ip=get_remote_address(request))
        redirect = RedirectResponse(url="/login", status_code=302)
        redirect.delete_cookie(key="access_token", path="/")
        redirect.delete_cookie(key="refresh_token", path="/api/auth")
        return redirect

No middleware de autenticação, após validar o JWT, verifique se o jti
está na blacklist antes de autorizar:

    jti = payload.get("jti")
    if jti:
        revoked = db.query(RevokedToken).filter(RevokedToken.jti == jti).first()
        if revoked:
            raise HTTPException(status_code=401, detail="Token revogado")

Adicione também um job de limpeza periódica que deleta tokens expirados
da tabela (podem ser removidos com segurança após expires_at):

    def cleanup_revoked_tokens(db: Session):
        db.query(RevokedToken).filter(
            RevokedToken.expires_at < datetime.utcnow()
        ).delete()
        db.commit()
