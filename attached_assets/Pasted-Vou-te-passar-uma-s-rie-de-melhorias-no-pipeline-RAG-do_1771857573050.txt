def _get_temperature(self, categoria: str, config: dict) -> float:
    """Temperatura adaptiva por tipo de resposta."""
    config_temp = config.get("temperature", 0.7) if config else 0.7
    
    # Se admin configurou manualmente, respeita (mas avisa se alto para DOCUMENTAL)
    if config and "temperature" in config:
        return config_temp
    
    # Defaults inteligentes por categoria
    temperature_map = {
        "DOCUMENTAL": 0.2,   # Dados factuais — precisão máxima
        "ESCOPO":     0.3,   # Técnico — quase sem variação
        "MERCADO":    0.4,   # Dados externos — leve flexibilidade
        "PITCH":      0.7,   # Texto criativo — variação necessária
        "SAUDACAO":   0.5,   # Conversacional — natural mas controlado
    }
    return temperature_map.get(categoria, 0.4)  # default 0.4 para demais

def _prepare_text_for_embedding(self, content: str, metadata: dict) -> str:
    """Enriquece o texto com contexto antes de embedar."""
    parts = []
    
    # Prefixo com identidade do produto (âncora semântica)
    if metadata.get("product_ticker"):
        parts.append(metadata["product_ticker"])
    if metadata.get("product_name") and metadata.get("product_name") != metadata.get("product_ticker"):
        parts.append(metadata["product_name"])
    if metadata.get("gestora"):
        parts.append(metadata["gestora"])
    if metadata.get("category"):
        parts.append(metadata["category"])
    if metadata.get("topic"):
        parts.append(metadata["topic"])
    
    prefix = " | ".join(parts)
    
    if prefix:
        return f"{prefix}\n{content}"
    return content
